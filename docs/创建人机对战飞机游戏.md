```typescript
// AI移动决策
function aiMoveDecision(ai: Player, bullets: Bullet[], enemies: Enemy[]) {
  // 计算危险区域（附近有子弹的区域）
  const dangerZones = bullets
    .filter(b => b.y > ai.y - 100) // 只考虑下方的子弹
    .map(b => ({ x: b.x, y: b.y, radius: 50 }));
  
  // 计算安全移动方向
  let safeDirection = { x: 0, y: 0 };
  
  // 如果附近有子弹，躲避
  if (dangerZones.length > 0) {
    // 找到最近的危险区域
    const nearestDanger = dangerZones.reduce((nearest, zone) => {
      const dist = distance(ai, zone);
      return dist < nearest.dist ? { zone, dist } : nearest;
    }, { zone: dangerZones[0], dist: Infinity });
    
    // 向相反方向移动
    safeDirection = {
      x: ai.x - nearestDanger.zone.x,
      y: ai.y - nearestDanger.zone.y
    };
  }
  
  // 添加随机移动，避免被预测
  const randomMove = {
    x: (Math.random() - 0.5) * 2,
    y: (Math.random() - 0.5) * 0.5 // 主要左右移动，少量上下
  };
  
  // 合并移动方向
  const finalMove = {
    x: safeDirection.x + randomMove.x,
    y: safeDirection.y + randomMove.y
  };
  
  return finalMove;
}
```

```typescript
function aiChargeDecision(ai: Player, gameState: GameState) {
  const chargeLevel = getChargeLevel(ai.chargeValue);
  
  // 根据难度调整决策
  const chargeThreshold = {
    easy: 0.8,    // 80%概率不蓄力
    normal: 0.5,  // 50%概率蓄力
    hard: 0.2     // 20%概率不蓄力
  };
  
  // 如果蓄力值达到满值，立即释放
  if (ai.chargeValue >= 90) {
    return 'release';
  }
  
  // 如果蓄力值达到3级，根据战场情况决定
  if (ai.chargeValue >= 60) {
    // 检查敌方是否有己方Boss
    const hasEnemyBoss = gameState.enemyScreen.hasBoss;
    if (!hasEnemyBoss && Math.random() > chargeThreshold[ai.difficulty]) {
      return 'continue_charge'; // 继续蓄力到满值
    } else {
      return 'release'; // 释放3级技能
    }
  }
  
  // 如果蓄力值达到2级，有概率释放
  if (ai.chargeValue >= 40) {
    if (Math.random() > chargeThreshold[ai.difficulty]) {
      return 'release';
    }
  }
  
  // 默认继续蓄力
  return 'continue_charge';
}
```

```typescript
function aiBombDecision(ai: Player, bullets: Bullet[], boss: Boss | null) {
  // 如果没有炸弹，不使用
  if (ai.bombs <= 0) return false;
  
  // 计算附近子弹密度
  const nearbyBullets = bullets.filter(b => 
    distance(b, ai) < 100
  ).length;
  
  // 根据难度调整阈值
  const bulletThreshold = {
    easy: 15,    // 15颗子弹以上才使用
    normal: 10,  // 10颗子弹以上
    hard: 5      // 5颗子弹以上
  };
  
  // 如果附近子弹密度高，使用炸弹
  if (nearbyBullets >= bulletThreshold[ai.difficulty]) {
    return true;
  }
  
  // 如果有敌方Boss且血量低，使用炸弹
  if (boss && ai.health < 30) {
    return true;
  }
  
  return false;
}
```

```typescript
class ParticleSystem {
  createExplosion(x: number, y: number, color: string) {
    for (let i = 0; i < 20; i++) {
      const angle = (Math.PI * 2 / 20) * i;
      const speed = 2 + Math.random() * 3;
      particles.push({
        x, y,
        vx: Math.cos(angle) * speed,
        vy: Math.sin(angle) * speed,
        life: 1.0,
        color
      });
    }
  }
  
  update(deltaTime: number) {
    for (let p of particles) {
      p.x += p.vx;
      p.y += p.vy;
      p.life -= deltaTime / 1000;
    }
    particles = particles.filter(p => p.life > 0);

}
```
